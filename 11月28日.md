```go
func search(nums []int, target int) int {
    r,l := len(nums),0
    for l<r{
        if nums[(l+r)/2] == target{
            return (l+r)/2
        } else if nums[(l+r)/2] > target{
            r=(l+r)/2
        } else {
            l=(l+r)/2 + 1
        }
    }
    return -1
}
```
