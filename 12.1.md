#### [19.删除链表的倒数第N个节点](https://leetcode.cn/problems/remove-nth-node-from-end-of-list/)

```go
func removeNthFromEnd(head *ListNode, n int) *ListNode {

  //使用虚拟头结点，方便处理删除实际头结点的逻辑

  dummy:=&ListNode{0,head}

  fast:=dummy

  slow:=dummy

  //快指针先走n+1步，便于删除操作

  for i:=0;i<=n;i++ {

    fast=fast.Next

  }

  //快慢指针同时移动，直到fast指向末尾

  for fast!=nil {

    fast=fast.Next

    slow=slow.Next

  }

  slow.Next=slow.Next.Next

  return dummy.Next

}
```
