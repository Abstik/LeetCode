func getFinalState(nums []int, k int, multiplier int) []int {
    min := 0
    length := len(nums)
    for k > 0 {
        for i := 0; i < length; i++ {
            if nums[min] > nums[i] {
                min = i
            }
        }
        nums[min] *= multiplier
        k--
        min = 0
    }
    return nums
}
